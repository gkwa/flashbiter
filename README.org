** wrapper for shell
*** powershell

#+begin_example
function flashbiterwrapper {
    $path = & 'flashbiter' $args
    $clean = $path -replace "`n", ""
    if ([string]::IsNullOrEmpty($clean)) {
        return
    }
    if (Test-Path -Path $clean -PathType Container) {
        Set-Location $clean
        git log --pretty=format:'%Creset%h%C(yellow)%d%Creset %ar %s' --reverse
    }
    Write-Output "$clean"
}
#+end_example
